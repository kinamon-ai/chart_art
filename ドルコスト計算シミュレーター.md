function calculateDCAResults(priceVariations) {
		const initialPrice = 1000; // 初期価格
		const investmentAmount = 10000; // 毎回の積立金額let results = [];
    let totalShares = 0;
    let totalInvestment = 0;

    for (let i = 0; i < priceVariations.length; i++) {
        // 価格変動率から現在価格を計算
        const currentPrice = initialPrice * (1 + priceVariations[i] / 100);

        // 今回購入できる株数
        const shares = investmentAmount / currentPrice;

        // 累計
        totalShares += shares;
        totalInvestment += investmentAmount;

        // 現在の評価額
        const currentValue = totalShares * currentPrice;

        // 損益率
        const profitLossRate = ((currentValue / totalInvestment) - 1) * 100;

        results.push({
            price: currentPrice,
            shares,
            totalShares,
            investment: investmentAmount,
            totalInvestment,
            value: currentValue,
            profitLossRate
        });
    }

    return results;
}

// 結果の計算と表示
function calculateAndDisplayResults() {
    // 価格変動リストを作成
    const priceVariations = calculatePriceVariations();

    // DCAシミュレーション
    const dcaResults = calculateDCAResults(priceVariations);

    // 結果グラフ表示
    displayResultChart(dcaResults);

    // サマリー表示
    displaySummary(dcaResults);

    // 共有ボタン表示
    shareButtonsContainer.style.display = 'flex';
}

// 結果グラフの表示
function displayResultChart(results) {
    const labels = results.map((_, index) => index + 1);
    const profitLossData = results.map(r => r.profitLossRate);

    // 棒グラフの色（損益によって変更）
    const backgroundColors = profitLossData.map(value =>
        value >= 0 ? 'rgba(39, 174, 96, 0.7)' : 'rgba(231, 76, 60, 0.7)'
    );

    const borderColors = profitLossData.map(value =>
        value >= 0 ? 'rgba(39, 174, 96, 1)' : 'rgba(231, 76, 60, 1)'
    );

    // 以前のチャートがあれば破棄
    if (resultChart) {
        resultChart.destroy();
    }

    // 新しいチャート作成
    resultChart = new Chart(resultCanvas, {
        type: 'bar',
        data: {
            labels: labels,
            datasets: [{
                label: '損益率 (%)',
                data: profitLossData,
                backgroundColor: backgroundColors,
                borderColor: borderColors,
                borderWidth: 1
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            scales: {
                y: {
                    beginAtZero: false,
                    grid: {
                        color: 'rgba(200, 200, 200, 0.2)'
                    },
                    ticks: {
                        callback: function(value) {
                            return value + '%';
                        }
                    }
                },
                x: {
                    grid: {
                        display: false
                    },
                    ticks: {
                        maxTicksLimit: 10
                    }
                }
            },
            plugins: {
                legend: {
                    display: false
                },
                tooltip: {
                    callbacks: {
                        label: function(context) {
                            const index = context.dataIndex;
                            const result = results[index];

                            return [
                                `損益率: ${result.profitLossRate.toFixed(2)}%`,
                                `積立回数: ${index + 1}回目`,
                                `価格: ${result.price.toFixed(2)}円`,
                                `評価額: ${result.value.toFixed(0)}円`,
                                `投資総額: ${result.totalInvestment.toFixed(0)}円`
                            ];
                        }
                    }
                }
            }
        }
    });
}

// サマリー情報の表示
function displaySummary(results) {
    const lastResult = results[results.length - 1];

    // 平均購入価格の計算
    const avgPrice = lastResult.totalInvestment / lastResult.totalShares;

    // 損益の計算
    const profitLoss = lastResult.value - lastResult.totalInvestment;

    // サマリー表示
    totalInvestmentEl.textContent = `${lastResult.totalInvestment.toLocaleString()}円`;
    finalValueEl.textContent = `${lastResult.value.toLocaleString()}円`;

    // 損益率表示（色分け）
    const plText = `${profitLoss.toLocaleString()}円 (${lastResult.profitLossRate.toFixed(2)}%)`;
    totalProfitLossEl.textContent = plText;
    totalProfitLossEl.className = 'summary-value ' + (profitLoss >= 0 ? 'positive' : 'negative');

    // 最終損益額の表示（色分け）
    absoluteProfitLossEl.textContent = `${profitLoss.toLocaleString()}円`;
    absoluteProfitLossEl.className = 'summary-value ' + (profitLoss >= 0 ? 'positive' : 'negative');

    averageBuyPriceEl.textContent = `${avgPrice.toFixed(2)}円`;

    // サマリーを表示
    summaryContainer.style.display = 'block';
}

// 共有機能
function getShareText() {
    const profitLossEl = document.getElementById('absoluteProfitLoss');
    const profitLoss = profitLossEl.textContent;
    const text = `ドルコスト平均法シミュレータで投資をシミュレーションしました！\\n最終損益: ${profitLoss}\\nあなたも試してみませんか？`;
    return text;
}

function getShareUrl() {
    return window.location.href;
}

function shareTwitter(e) {
    e.preventDefault();
    const text = getShareText();
    const url = getShareUrl();
    const shareUrl = `https://twitter.com/intent/tweet?text=${encodeURIComponent(text)}&url=${encodeURIComponent(url)}`;
    window.open(shareUrl, '_blank', 'width=550,height=420');
}

function shareLine(e) {
    e.preventDefault();
    const text = getShareText();
    const url = getShareUrl();
    const shareUrl = `https://line.me/R/msg/text/?${encodeURIComponent(text + '\\n' + url)}`;
    window.open(shareUrl, '_blank');
}

function shareFacebook(e) {
    e.preventDefault();
    const url = getShareUrl();
    const shareUrl = `https://www.facebook.com/sharer/sharer.php?u=${encodeURIComponent(url)}`;
    window.open(shareUrl, '_blank', 'width=550,height=420');
}

function shareInstagram(e) {
    e.preventDefault();
    // Instagramは直接的なWeb共有APIがないため、クリップボードにコピー
    const text = getShareText();
    const url = getShareUrl();
    const fullText = `${text}\\n${url}`;

    navigator.clipboard.writeText(fullText).then(() => {
        alert('テキストをクリップボードにコピーしました！\\nInstagramアプリを開いて投稿に貼り付けてください。');
    }).catch(() => {
        alert('クリップボードへのコピーに失敗しました。');
    });
}
